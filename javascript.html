JS Basics Refresher

Node.js is built on JavaScript — mastering JS fundamentals is essential before diving into Node-specific concepts.

Variables & Data Types

// var (function-scoped, hoisted) — avoid in modern JS
        var x = 10;
        
        // let (block-scoped, not hoisted)
        let name = "Alice";
        
        // const (block-scoped, immutable binding)
        const PI = 3.14159;
        
        // Primitive types
        let num = 42;            // number
        let str = "hello";       // string
        let bool = true;         // boolean
        let nothing = null;      // null
        let undef = undefined;   // undefined
        let sym = Symbol("id");  // symbol
        let big = 9007199254740993n; // bigint
        
        // Reference types
        let arr = [1, 2, 3];
        let obj = { key: "value" };
        let fn = function() {};

Type Coercion & Equality

// == (loose equality — coerces types)
        0 == false    // true
        "" == false   // true
        null == undefined // true
        
        // === (strict equality — no coercion)
        0 === false   // false
        "1" === 1     // false
        
        // Common gotchas
        typeof null   // "object" (historical bug)
        typeof NaN    // "number"
        NaN === NaN   // false — use Number.isNaN(x)
        
        // Truthy / Falsy
        // Falsy: false, 0, "", null, undefined, NaN, 0n
        // Truthy: everything else (including [], {}, "0")

Functions

// Function declaration (hoisted)
        function greet(name) {
          return `Hello, ${name}!`;
        }
        
        // Function expression (not hoisted)
        const add = function(a, b) {
          return a + b;
        };
        
        // Arrow function (no own `this`, no `arguments`)
        const multiply = (a, b) => a * b;
        
        // Default parameters
        function connect(host = "localhost", port = 3000) {
          return `${host}:${port}`;
        }
        
        // Rest parameters
        function sum(...nums) {
          return nums.reduce((a, b) => a + b, 0);
        }
        
        // Spread operator
        const nums = [1, 2, 3];
        console.log(Math.max(...nums)); // 3
Key Rule: Arrow functions inherit this from their enclosing lexical scope. Regular functions get their own this bound at call time.

Closures & Scope

A closure is a function that retains access to its lexical scope even when invoked outside that scope.

Closure in Action

function makeCounter(start = 0) {
          let count = start; // enclosed variable
        
          return {
            increment() { return ++count; },
            decrement() { return --count; },
            value()     { return count; },
          };
        }
        
        const counter = makeCounter(10);
        counter.increment(); // 11
        counter.increment(); // 12
        counter.decrement(); // 11
        counter.value();     // 11
        
        // `count` is NOT accessible from outside
        console.log(counter.count); // undefined

Classic var-in-loop Bug

// BUG: all callbacks share the same `i`
        for (var i = 0; i < 3; i++) {
          setTimeout(() => console.log(i), 100); // 3, 3, 3
        }
        
        // FIX 1: use let (block-scoped per iteration)
        for (let i = 0; i < 3; i++) {
          setTimeout(() => console.log(i), 100); // 0, 1, 2
        }
        
        // FIX 2: IIFE to capture current value
        for (var i = 0; i < 3; i++) {
          (function(j) {
            setTimeout(() => console.log(j), 100); // 0, 1, 2
          })(i);
        }

Practical Closure Patterns

// Module pattern (pre-ES6)
        const cache = (function() {
          const store = {};
          return {
            get: (key) => store[key],
            set: (key, val) => { store[key] = val; },
            has: (key) => key in store,
          };
        })();
        
        // Memoization
        function memoize(fn) {
          const cache = new Map();
          return function(...args) {
            const key = JSON.stringify(args);
            if (cache.has(key)) return cache.get(key);
            const result = fn.apply(this, args);
            cache.set(key, result);
            return result;
          };
        }
        
        const expensiveCalc = memoize((n) => {
          // simulate expensive computation
          return n * n;
        });
        expensiveCalc(5); // computed
        expensiveCalc(5); // cached

Prototypes & Inheritance

Prototype Chain

// Every object has a [[Prototype]] (accessible via __proto__ or Object.getPrototypeOf)
        const animal = {
          speak() { return `${this.name} makes a sound`; }
        };
        
        const dog = Object.create(animal);
        dog.name = "Rex";
        dog.speak(); // "Rex makes a sound" — found on prototype
        
        // Prototype chain: dog → animal → Object.prototype → null

Classes (ES6 Syntactic Sugar)

class Shape {
          constructor(color) {
            this.color = color;
          }
        
          describe() {
            return `A ${this.color} shape`;
          }
        
          static create(color) {        // static method — called on class, not instance
            return new Shape(color);
          }
        }
        
        class Circle extends Shape {
          #radius; // private field (ES2022)
        
          constructor(color, radius) {
            super(color);
            this.#radius = radius;
          }
        
          area() {
            return Math.PI * this.#radius ** 2;
          }
        
          get radius() { return this.#radius; }
          set radius(r) {
            if (r <= 0) throw new Error("Radius must be positive");
            this.#radius = r;
          }
        
          describe() {
            return `${super.describe()} (circle, r=${this.#radius})`;
          }
        }
        
        const c = new Circle("red", 5);
        c.area();     // 78.53...
        c.describe(); // "A red shape (circle, r=5)"
Interview Point: Classes in JS are NOT like Java classes. They are syntactic sugar over the prototype chain. typeof Circle === "function" is true.

Async JS: Callbacks → Promises → Async/Await

Callbacks (Oldest Pattern)

const fs = require("fs");
        
        // Error-first callback convention (Node.js style)
        fs.readFile("data.txt", "utf8", (err, data) => {
          if (err) return console.error("Error:", err);
          console.log(data);
        });
        
        // Callback hell — deeply nested, hard to read
        getUser(id, (err, user) => {
          if (err) return handleError(err);
          getOrders(user.id, (err, orders) => {
            if (err) return handleError(err);
            getOrderDetails(orders[0].id, (err, details) => {
              // pyramid of doom...
            });
          });
        });

Promises (ES6)

// Creating a promise
        const delay = (ms) => new Promise((resolve) => setTimeout(resolve, ms));
        
        // Promise chaining (flat, readable)
        fetchUser(id)
          .then((user) => fetchOrders(user.id))
          .then((orders) => fetchDetails(orders[0].id))
          .then((details) => console.log(details))
          .catch((err) => console.error(err))
          .finally(() => console.log("done"));
        
        // Promise combinators
        Promise.all([p1, p2, p3])      // resolves when ALL resolve; rejects on first rejection
        Promise.allSettled([p1, p2])   // waits for ALL; returns [{status, value/reason}]
        Promise.race([p1, p2])         // settles with the FIRST settled promise
        Promise.any([p1, p2])          // resolves with FIRST resolved; rejects if ALL reject
        
        // Example: parallel fetch with timeout
        const withTimeout = (promise, ms) =>
          Promise.race([promise, delay(ms).then(() => { throw new Error("Timeout"); })]);
        
        const results = await Promise.all([
          withTimeout(fetch("/api/users"), 5000),
          withTimeout(fetch("/api/orders"), 5000),
        ]);

Async/Await (ES2017)

// async functions always return a Promise
        async function fetchUserData(id) {
          try {
            const user = await fetchUser(id);      // suspends here
            const orders = await fetchOrders(user.id);
            return { user, orders };
          } catch (err) {
            throw new Error(`Failed to fetch user data: ${err.message}`);
          }
        }
        
        // Parallel execution — DON'T await sequentially if independent
        // ❌ Sequential (slow — waits for each before starting next)
        const user = await fetchUser(1);
        const settings = await fetchSettings(1);
        
        // ✅ Parallel (fast — both kick off simultaneously)
        const [user, settings] = await Promise.all([fetchUser(1), fetchSettings(1)]);
        
        // Top-level await (ES2022, ESM only)
        const config = await loadConfig(); // valid at module top level in .mjs
Important: async/await is syntactic sugar over Promises. Under the hood, the JS engine converts await into .then() chains. Understanding Promises is essential for debugging async code.

JavaScript Interview Q&A

Q: What is the difference between == and ===?
== (loose equality) performs type coercion before comparing. === (strict equality) compares both value AND type without coercion. Always prefer === to avoid unexpected coercion bugs (e.g., 0 == false is true but 0 === false is false).
Q: Explain hoisting in JavaScript.
Hoisting is JavaScript's behavior of moving declarations to the top of their scope before execution. var is hoisted and initialized to undefined. function declarations are fully hoisted (both declaration and body). let/const are hoisted but NOT initialized — accessing them before declaration throws a ReferenceError (Temporal Dead Zone). class declarations are also in the TDZ.
Q: What is the Temporal Dead Zone (TDZ)?
The period between entering a scope and the let/const declaration being evaluated. Accessing a variable in the TDZ throws a ReferenceError. This prevents the "use before initialization" bugs that var allows.
Q: What is a closure? Provide a real-world use case.
A closure is a function that retains access to variables from its outer scope even after that scope has finished executing. Real use cases: (1) data encapsulation / module pattern — hide private state; (2) function factories — makeMultiplier(3) returns a function that always multiplies by 3; (3) memoization — cache results using an enclosed Map; (4) event handlers that need context from when they were created.
Q: Explain this in JavaScript — when does it change?
this depends on how a function is called: (1) Global context: window in browser, global in Node (or undefined in strict mode). (2) Method call: obj.fn() — this is obj. (3) Constructor: new Fn() — this is the new instance. (4) Explicit binding: fn.call(ctx), fn.apply(ctx), fn.bind(ctx). (5) Arrow functions: inherit this lexically from enclosing scope, cannot be rebound.
Q: What is the difference between null and undefined?
undefined means a variable has been declared but not assigned. null is an intentional absence of value — it must be explicitly assigned. typeof undefined === "undefined", typeof null === "object" (historical bug). Use ?? (nullish coalescing) to handle both: val ?? "default".
Q: Explain prototypal inheritance vs classical inheritance.
Classical inheritance (Java, C++) uses classes as blueprints — objects are instances of classes, hierarchy is fixed at compile time. Prototypal inheritance (JavaScript) uses objects directly — every object has a [[Prototype]] link to another object. Property lookup traverses this chain at runtime. ES6 class syntax is purely syntactic sugar over the prototype chain — typeof MyClass === "function". This makes JS more flexible: you can modify prototypes at runtime, use mixins, or compose objects directly with Object.create().
Q: What is event delegation?
Instead of attaching event listeners to each child element, attach one listener to a parent and use event.target to determine which child was clicked. Benefits: (1) fewer listeners = better memory performance; (2) works for dynamically added children; (3) simplifies listener management. Used heavily in virtual DOM frameworks and large lists.
Q: How does Promise.all differ from Promise.allSettled?
Promise.all rejects immediately when ANY promise rejects (fail-fast). Promise.allSettled waits for ALL promises to settle (resolve or reject) and returns an array of { status: 'fulfilled', value } or { status: 'rejected', reason } objects. Use allSettled when you need results from all operations even if some fail (e.g., batch operations where partial failure is acceptable).
Q: What is the difference between call, apply, and bind?
All three set this explicitly. fn.call(ctx, arg1, arg2) invokes immediately with individual args. fn.apply(ctx, [arg1, arg2]) invokes immediately with an array of args. fn.bind(ctx, arg1) returns a NEW function with this permanently bound — doesn't invoke. Useful for partial application and creating callbacks with correct context.
